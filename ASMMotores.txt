#include "p16f887.inc"

; CONFIG1
; __config 0xE0F4
 __CONFIG _CONFIG1, _FOSC_INTRC_NOCLKOUT & _WDTE_OFF & _PWRTE_OFF & _MCLRE_ON & _CP_OFF & _CPD_OFF & _BOREN_OFF & _IESO_OFF & _FCMEN_OFF & _LVP_OFF
; CONFIG2
; __config 0xFFFF
 __CONFIG _CONFIG2, _BOR4V_BOR40V & _WRT_OFF
;
;*****************************************************************

; TODO INSERT CONFIG HERE

;=================================================================
;Definicion de variables
;=================================================================


	GPR_VAR		UDATA
	ADC		RES	1
	PWM_TEMP	RES	1
	PWM		RES 	1
	LSB_PWM		RES	1
	MSB_PWM		RES	1
	W_TEMP		RES	1	; w register for context saving (ACCESS)
	STATUS_TEMP	RES	1
	
;*****************************************************************
; Reset vector
;*****************************************************************

RES_VECT	CODE	0x0000	; processor rest vector
    GOTO	START		; go to beginning of program

MAIN_PROG CODE			; let linker place main program

;==================================================================
SET_PWM MACRO CCPxCON, DCxB0, DCxB1, CCPRxL
	BCF STATUS, C		; LIMPIO BANDERA DE ACARREO
	RRF PWM_TEMP, F		; CORRO A LA IZQUIERDA LOS BITS
	; MOVER C A CCPxCON, DCxB0
	BTFSS STATUS, C
	BSF CCPxCON, DCxB0
	BTFSC STATUS, C
	BCF CCPxCON, DCxB0

	BCF STATUS, C		; LIMPIO NUEVAMENTE BANDERA DE ACARREO
	RRF PWM_TEMP, F		; CORRO A LA IZQUIERDA LOS BITS
	; MOVER C A CCPxCON, DCxB1
	BTFSS STATUS, C
	BSF CCPxCON, DCxB1
	BTFSC STATUS, C
	BCF CCPxCON, DCxB1

	; MUEVO LOS BITS RESTANTES A CCPRxL
	MOVF PWM_TEMP, W
	MOVWF CCPRxL
ENDM
;==================================================================
START 

SETUP:
	BSF STATUS, RP0
	BCF STATUS, RP1	; BANCO 1

	CLRF TRISD
	
	;OSCILADOR===========================
	BCF OSCCON, IRCF2
	BSF OSCCON, IRCF1
	BSF OSCCON, IRCF0	; FRECUENCIA 500KHZ

	BCF OSCCON, OSTS	; RELOJ INTERNO DE INICIO

	BSF OSCCON, HTS		;RELOJ HTS ESTABLE
	BSF OSCCON, SCS		;OSCILADOR INTERNO

	;ADC======================================
	BSF TRISA, RA0		; POTENCIOMETRO EN RA0
	BSF TRISB, RB2

	BCF ADCON1, VCFG1	;REFERENCIA -VSS
	BCF ADCON1, VCFG0	;REFERENCIA +VDD

	BCF ADCON1, ADFM	;JUSTIFICACION IZQUIERDA

	BSF PIE1, ADIE		;HABILITO INTERRUPCION POR ADC

	BSF STATUS, RP1		;BANCO3

	CLRF ANSEL
	CLRF ANSELH		;ESTABLEZCO QUE TODAS LAS ENTRADAS SON DIGITALES

	BSF ANSEL, ANS0		;RA0 COMO ENTRADA ANALOGICA
	BSF ANSEL, ANS8		;RB2 COMO ENTRADA ANALOGICA

	BCF STATUS, RP0
	BCF STATUS, RP1		;BANCO 0

	BSF ADCON0, ADCS1
	BSF ADCON0, ADCS1	;FRC

	;RA0 ES CANAL AN0
	BCF ADCON0, CHS3
	BCF ADCON0, CHS2
	BCF ADCON0, CHS1
	BCF ADCON0, CHS0	;CANAL ANALOGICO 0 
	
	BSF ADCON0, ADON	;ENCIENDO EL MODULO ADC

	BCF PIR1, ADIF		;BORRO BANDERA INTERRUPCION

	;PWM=================================================
	
	BSF STATUS, RP0		;BANCO 1
	BSF TRISC, RC1		;SALIDA PWM2 LO COLOCO COMO ENTRADA
	BSF TRISC, RC2		;SALIDA PWM1 LO COLOCO COMO ENTRADA

	MOVLW .155
	MOVWF PR2		;PERIODO DE SEÑAL PWM2

	BCF STATUS, RP0		;BANCO 0

	BSF CCP2CON, CCP2M3
	BSF CCP2CON, CCP2M2
	BSF CCP2CON, CCP2M1
	BSF CCP2CON, CCP2M0	;MODO PWM2

	BCF CCP1CON, P1M1
	BCF CCP1CON, P1M0	;SINGLE OUTPUT PWM2

	BSF CCP1CON, CCP1M3
	BSF CCP1CON, CCP1M2
	BCF CCP1CON, CCP1M1
	BCF CCP1CON, CCP1M0	;MODO PWM1

	BSF CCP2CON, DC2B1
	BSF CCP2CON, DC2B0	; COLOCO LOS 31

	BSF CCP1CON, DC1B1
	BSF CCP1CON, DC1B0	; COLOCO LOS 31

	MOVLW B'00000111'
	MOVWF CCPR2L		;BITS MAS SIGNIFICATIVOS DE 31

	MOVLW B'00000111'
	MOVWF CCPR1L		;BITS MAS SIGNIFICATIVOS DE 31

	;TMR2=================================================
	BCF PIR1, TMR2IF	;BORRO BANDERA TMR2
		
	BSF T2CON, T2CKPS1
	BSF T2CON, T2CKPS0	;PRESCALER 16

	BSF T2CON, TMR2ON	;HABILITO TMR2

CHECKTMR2:
	BTFSS PIR1, TMR2IF
	GOTO CHECKTMR2

	BCF PIR1, TMR2IF

	BSF STATUS, RP0		;BANCO 1

	BCF TRISC, RC1		;BORRO BANDERA DE INTERRUPCION
	BCF TRISC, RC2

	BCF STATUS, RP0		;BANCO 0
	
	;BSF INTCON, GIE	;HABILITO TODAS LAS INTERRUPCIONES
	CLRF PORTD
LOOP:
;=================================================================
;VERIFICO QUE PIN ANALOGICO ESTE ENVIANDO SEÑALES
	BTFSS ADCON0, CHS3
	MOVLW B'0'
	BTFSS ADCON0, CHS3
	MOVLW B'1'
	MOVWF PWM
	
	;CAMBIO DE ENTRADA ANALOGICA
	BTFSC PWM, 0
	BCF ADCON0, CHS3
	BTFSS PWM, 0
	BCF ADCON0, CHS3
	;=========================================================
	
	BSF ADCON0, ADON
	BSF ADCON0, GO		;EMPEZAMOS LA CONVERSION
	
	 CHECKADC:
	BTFSC ADCON0, GO	;VERIFICO SI TERMINO DE REALIZAR LA CONVERSION
	GOTO CHECKADC
	
	BCF PIR1, ADIF		;BORRO LA BANDERA DE INTERRUPCIONES
	
	CLRF PWM_TEMP 		;REINICIO VALOR DE PWM
	
	MOVF ADRESH, W
	MOVWF ADC		;VALOR QUE DICTA ANCHO DE PULSO
	
	BCF STATUS, C 		;LIMPIO BANDERA DE ACARREO
	MOVLW .3		;DENOMINADOR DE DIVISION
	SUBWF ADC, F		;RESTA DE NUMERADOR Y DENOMINADOR
	BTFSC STATUS, C	
	INCF PWM_TEMP		;INCREMENTO RESIDUO
	BTFSC STATUS, C
	GOTO $-6		;REPITO SI AUN SE PUEDE DIVIDIR
	
	MOVLW .16		;BASE DE ANCHO DE PULSO
	ADDWF PWM_TEMP, F	;ANCHO DE PULSO A CARGAR
	
	BTFSC PWM, 0		;MOVER UN SERVO
	GOTO PWM1
	SET_PWM CCP2CON, DC2B0, DC2B1, CCPR2L
	
	GOTO LOOP

PWM1:		;MOVER OTRO SERVO
	SET_PWM CCP1CON, DC1B0, DC1B1, CCPR1L
	MOVF PWM_TEMP, W
	MOVWF PORTD
	
	GOTO LOOP
	
	END
